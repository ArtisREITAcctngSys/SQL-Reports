//Select Header
select distinct
p.scode pcode,
p.saddr1 pname,
Convert(Varchar,'#asofdt#',101) asofdt

from property p
where 1=1
#condition1#
//End Select

//Select Dataset1
select 
p.scode pcode,
u.scode ucode,
ut.scode unittype,
u.DSQFT unitarea,
isnull(t.scode,'Vacant') tcode,
isnull((t.SFIRSTNAME + ' ' + t.slastname),'Vacant') tenantname,
u.SRENT marketrent,
t.srent actualrent,
t.SDEPOSIT0 deposit,
isnull(t.sdeposit1,0) + isnull(t.SDEPOSIT2,0) + isnull(t.SDEPOSIT3,0) otherdeposit,
t.dtmovein movein,
t.dtleaseto leaseexpiration,
t.dtmoveout moveoutdate,
case when isnull(b.hmy,0) = 0 then 'No' else 'Yes' end PAD,
t.hmyperson thmy
from property p
inner join unit u on u.hproperty = p.hmy
inner join unittype ut on ut.hmy = u.hunittype
left outer join tenant t on u.hmy = t.hunit and t.DTLEASETO >= '#asofdt#' and t.dtleasefrom <= '#asofdt#' and isnull(t.dtmoveout,'12/31/2100') >=case when '#moveout#' = 'Include' then '#asofdt#' else DATEADD(Month,1,'#asofdt#') end
Left outer join achdata b on b.hperson = t.hmyperson

where 1=1 
#condition1#
and case when  isnull(b.hmy,0) = 0 then 'No' else 'Yes' end like (case when '#PAD#' = 'Show All' then '%'  when '#PAD#' = 'Only PAD' then 'Yes' when '#PAD#'  = 'Exclude PAD' then 'No' end)
order by 2
//End Select

/Columns
//Type,  Name, Hd1, Hd2,            Hd3,          Hd4, Show,  Clr, Frmla, Drill,  Key, Width
  //End columns

//end filter
